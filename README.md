          `🌐 From: nobellocksmith.com`;
        
        try {
          const telegramResponse = await fetch(`https://api.telegram.org/bot${TELEGRAM_BOT_TOKEN}/sendMessage`, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
              chat_id: TELEGRAM_CHAT_ID,
              text: message,
              parse_mode: 'HTML'
            })
          });
          
          await fetch('/', {
            method: 'POST',
            headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
            body: new URLSearchParams(formData).toString()
          });
          
          if (telegramResponse.ok) {
            feedback.innerHTML = '<i class="fas fa-check-circle"></i> Thank you! We\'ll contact you as soon as possible.';

            form.reset();
            setTimeout(() => { feedback.innerHTML = ''; }, 5000);
          } else {
            throw new Error('Telegram notification failed');
          }
          
        } catch (error) {
          console.error('Error:', error);
          feedback.innerHTML = '<i class="fas fa-exclamation-circle"></i> Message sent! We\'ll contact you shortly.';
          feedback.style.color = "#C5A028";
          setTimeout(() => { feedback.innerHTML = ''; }, 6000);
